**Database Schema**

After migration, the database schema will be created with table names in plural form and low case snake case letters:

```sql
CREATE TABLE entity_types (
    id SERIAL PRIMARY KEY,
    name VARCHAR(255) NOT NULL
);

CREATE TABLE entities (
    id SERIAL PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    entity_type_id INTEGER NOT NULL REFERENCES entity_types(id),
    CONSTRAINT fk_entity_type FOREIGN KEY (entity_type_id) REFERENCES entity_types (id)
);

CREATE TABLE attributes (
    id SERIAL PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    entity_type_id INTEGER NOT NULL REFERENCES entity_types(id),
    CONSTRAINT fk_entity_type FOREIGN KEY (entity_type_id) REFERENCES entity_types (id)
);

CREATE TABLE values (
    id SERIAL PRIMARY KEY,
    value VARCHAR(255) NOT NULL,
    entity_id INTEGER NOT NULL REFERENCES entities(id),
    attribute_id INTEGER NOT NULL REFERENCES attributes(id),
    CONSTRAINT fk_entity FOREIGN KEY (entity_id) REFERENCES entities(id),
    CONSTRAINT fk_attribute FOREIGN KEY (attribute_id) REFERENCES attributes(id)
);
```

import { Component, OnInit } from '@angular/core';
import { AttributeService, Attribute } from '../../services/attribute.service';

@Component({
  selector: 'app-attribute',
  templateUrl: './attribute.component.html',
  styleUrls: ['./attribute.component.css']
})
export class AttributeComponent implements OnInit {
  attributes: Attribute[] = [];
  selectedAttribute?: Attribute;
  newAttribute: Attribute = { name: '', value: '' };

  constructor(private attributeService: AttributeService) { }

  ngOnInit(): void {
    this.loadAttributes();
  }

  loadAttributes() {
    this.attributeService.getAll().subscribe(data => {
      this.attributes = data;
    });
  }

  selectAttribute(attribute: Attribute) {
    this.selectedAttribute = { ...attribute };
  }

  saveAttribute() {
    if (this.selectedAttribute && this.selectedAttribute.id) {
      // Update
      this.attributeService.update(this.selectedAttribute).subscribe(() => {
        this.loadAttributes();
        this.selectedAttribute = undefined;
      });
    } else {
      // Create
      this.attributeService.create(this.newAttribute).subscribe(() => {
        this.loadAttributes();
        this.newAttribute = { name: '', value: '' };
      });
    }
  }

  deleteAttribute(id?: number) {
    if (id) {
      this.attributeService.delete(id).subscribe(() => {
        this.loadAttributes();
      });
    }
  }

  cancel() {
    this.selectedAttribute = undefined;
    this.newAttribute = { name: '', value: '' };
  }
}

import { Component, OnInit } from '@angular/core';
import { EntityTypeService, EntityType } from '../../services/entity-type.service';

@Component({
  selector: 'app-entity-type',
  templateUrl: './entity-type.component.html',
  styleUrls: ['./entity-type.component.css']
})
export class EntityTypeComponent implements OnInit {
  entityTypes: EntityType[] = [];
  selectedEntityType?: EntityType;
  newEntityType: EntityType = { name: '' };

  constructor(private entityTypeService: EntityTypeService) { }

  ngOnInit(): void {
    this.loadEntityTypes();
  }

  loadEntityTypes() {
    this.entityTypeService.getAll().subscribe(data => {
      this.entityTypes = data;
    });
  }

  selectEntityType(entityType: EntityType) {
    this.selectedEntityType = { ...entityType };
  }

  saveEntityType() {
    if (this.selectedEntityType && this.selectedEntityType.id) {
      // For this API, create is only POST, so editing might need a PUT method if available.
      // Assuming create only, so editing is not implemented here.
    } else {
      this.entityTypeService.create(this.newEntityType).subscribe(() => {
        this.loadEntityTypes();
        this.newEntityType = { name: '' };
      });
    }
  }

  deleteEntityType(id?: number) {
    if (id) {
      this.entityTypeService.delete(id).subscribe(() => {
        this.loadEntityTypes();
      });
    }
  }

  cancel() {
    this.selectedEntityType = undefined;
    this.newEntityType = { name: '' };
  }
}
